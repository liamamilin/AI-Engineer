31  Chapter 26 — Cost Optimization Strategies

31.1 Introduction

Chapter 16 identified cost as one of the three fundamental constraints in scaling AI systems, alongside latency and reliability. While powerful models and sophisticated architectures deliver impressive results, they also generate substantial costs at scale.

This chapter provides practical strategies for optimizing AI system costs without sacrificing quality. It covers model selection and routing, caching techniques, resource optimization, monitoring and management, and real-world cost optimization patterns.

# src/cost/tracker.py
from typing import Dict
from datetime import datetime

class CostTracker:
    def __init__(self):
        self.costs = []
        self.budgets = {}
    
    def log_cost(self, component: str, operation: str, 
                amount: float, metadata: Dict = None):
        """Log a cost event."""
        self.costs.append({
            'component': component,
            'operation': operation,
            'amount': amount,
            'metadata': metadata or {},
            'timestamp': datetime.now()
        })
    
    def set_budget(self, component: str, budget: float, 
                  period: str = "daily"):
        """Set a budget for a component."""
        self.budgets[component] = {
            'budget': budget,
            'period': period,
            'start': datetime.now()
        }
    
    def check_budget(self, component: str) -> Dict:
        """Check if component is within budget."""
        if component not in self.budgets:
            return {'within_budget': True, 'spent': 0, 'budget': 0}
        
        budget_info = self.budgets[component]
        
        # Calculate spent in period
        spent = sum(
            c['amount'] for c in self.costs
            if c['component'] == component and
            self.in_period(c['timestamp'], budget_info['start'], budget_info['period'])
        )
        
        within_budget = spent <= budget_info['budget']
        
        return {
            'within_budget': within_budget,
            'spent': spent,
            'budget': budget_info['budget'],
            'remaining': budget_info['budget'] - spent,
            'utilization': spent / budget_info['budget'] if budget_info['budget'] > 0 else 0
        }
    
    def in_period(self, timestamp: datetime, start: datetime, 
                 period: str) -> bool:
        """Check if timestamp is within the budget period."""
        if period == "daily":
            return (timestamp - start).days < 1
        elif period == "weekly":
            return (timestamp - start).days < 7
        elif period == "monthly":
            return (timestamp - start).days < 30
        return True

31.3.1 Model Cost Analysis

# src/cost/model_analyzer.py

class ModelCostAnalyzer:
    def __init__(self):
        self.model_pricing = {
            'gpt-4': {
                'input_cost_per_1k': 0.03,
                'output_cost_per_1k': 0.06,
                'tokens_per_second': 10
            },
            'gpt-3.5-turbo': {
                'input_cost_per_1k': 0.0015,
                'output_cost_per_1k': 0.002,
                'tokens_per_second': 50
            },
            'gpt-4-turbo': {
                'input_cost_per_1k': 0.01,
                'output_cost_per_1k': 0.03,
                'tokens_per_second': 30
            }
        }
    
    def estimate_cost(self, model: str, input_tokens: int, 
                     output_tokens: int) -> float:
        """Estimate cost for a model call."""
        pricing = self.model_pricing.get(model)
        if not pricing:
            raise ValueError(f"Unknown model: {model}")
        
        input_cost = (input_tokens / 1000) * pricing['input_cost_per_1k']
        output_cost = (output_tokens / 1000) * pricing['output_cost_per_1k']
        
        return input_cost + output_cost
    
    def estimate_latency(self, model: str, total_tokens: int) -> float:
        """Estimate latency for a model call."""
        pricing = self.model_pricing.get(model)
        if not pricing:
            raise ValueError(f"Unknown model: {model}")
        
        return total_tokens / pricing['tokens_per_second']
    
    def compare_models(self, models: List[str], input_tokens: int,
                      output_tokens: int) -> Dict:
        """Compare costs and latency across models."""
        comparison = {}
        
        for model in models:
            comparison[model] = {
                'cost': self.estimate_cost(model, input_tokens, output_tokens),
                'latency': self.estimate_latency(model, input_tokens + output_tokens)
            }
        
        return comparison

31.3.2 Intelligent Model Routing

# src/cost/model_router.py

class ModelRouter:
    def __init__(self, model_analyzer, model_clients):
        self.model_analyzer = model_analyzer
        self.model_clients = model_clients
        self.routing_rules = []
    
    def add_routing_rule(self, condition: Callable, model: str):
        """Add a routing rule."""
        self.routing_rules.append({
            'condition': condition,
            'model': model
        })
    
    def route(self, input_data: Dict, default_model: str = "gpt-3.5-turbo") -> str:
        """Route request to appropriate model."""
        # Check routing rules
        for rule in self.routing_rules:
            if rule['condition'](input_data):
                return rule['model']
        
        # Use default model
        return default_model
    
    def execute_with_routing(self, input_data: Dict, 
                           default_model: str = "gpt-3.5-turbo") -> Dict:
        """Execute request with intelligent routing."""
        # Route to appropriate model
        model = self.route(input_data, default_model)
        
        # Execute with selected model
        client = self.model_clients[model]
        start_time = datetime.now()
        
        output = client.generate(input_data['prompt'])
        
        end_time = datetime.now()
        
        # Calculate cost
        input_tokens = len(input_data['prompt'].split())  # Approximate
        output_tokens = len(output.split())
        cost = self.model_analyzer.estimate_cost(
            model, input_tokens, output_tokens
        )
        
        return {
            'output': output,
            'model': model,
            'cost': cost,
            'latency': (end_time - start_time).total_seconds(),
            'input_tokens': input_tokens,
            'output_tokens': output_tokens
        }

# src/cost/cost_selector.py

class CostAwareModelSelector:
    def __init__(self, model_analyzer):
        self.model_analyzer = model_analyzer
    
    def select_cheapest_model(self, models: List[str], 
                             input_tokens: int, output_tokens: int,
                             min_quality: float = 0.8) -> str:
        """Select cheapest model that meets quality requirements."""
        # Filter models by quality
        qualified_models = [
            m for m in models 
            if self.get_model_quality(m) >= min_quality
        ]
        
        if not qualified_models:
            raise ValueError("No models meet quality requirements")
        
        # Select cheapest
        costs = {
            m: self.model_analyzer.estimate_cost(m, input_tokens, output_tokens)
            for m in qualified_models
        }
        
        return min(costs.items(), key=lambda x: x[1])[0]
    
    def get_model_quality(self, model: str) -> float:
        """Get quality score for a model."""
        # This would be based on historical performance
        quality_scores = {
            'gpt-4': 0.95,
            'gpt-4-turbo': 0.92,
            'gpt-3.5-turbo': 0.85
        }
        return quality_scores.get(model, 0.5)

31.4 Caching Strategies

31.4.1 Multi-Level Caching

# src/cost/cache.py
from typing import Optional, Dict
import hashlib
import json

class MultiLevelCache:
    def __init__(self):
        self.l1_cache = {}  # In-memory cache
        self.l2_cache = {}  # Persistent cache
        self.l1_size = 100
        self.l1_hits = 0
        self.l1_misses = 0
    
    def get_cache_key(self, prompt: str, model: str) -> str:
        """Generate cache key."""
        key_data = f"{model}:{prompt}"
        return hashlib.md5(key_data.encode()).hexdigest()
    
    def get(self, prompt: str, model: str) -> Optional[str]:
        """Get cached response."""
        key = self.get_cache_key(prompt, model)
        
        # Check L1 cache
        if key in self.l1_cache:
            self.l1_hits += 1
            return self.l1_cache[key]
        
        self.l1_misses += 1
        
        # Check L2 cache
        if key in self.l2_cache:
            self.l1_cache[key] = self.l2_cache[key]
            return self.l2_cache[key]
        
        return None
    
    def set(self, prompt: str, model: str, response: str):
        """Cache a response."""
        key = self.get_cache_key(prompt, model)
        
        # Add to L1 cache
        self.l1_cache[key] = response
        
        # Evict if necessary
        if len(self.l1_cache) > self.l1_size:
            oldest_key = next(iter(self.l1_cache))
            del self.l1_cache[oldest_key]
        
        # Add to L2 cache
        self.l2_cache[key] = response
    
    def get_stats(self) -> Dict:
        """Get cache statistics."""
        total = self.l1_hits + self.l1_misses
        hit_rate = self.l1_hits / total if total > 0 else 0
        
        return {
            'l1_hits': self.l1_hits,
            'l1_misses': self.l1_misses,
            'hit_rate': hit_rate,
            'l1_size': len(self.l1_cache),
            'l2_size': len(self.l2_cache)
        }

31.4.2 Semantic Caching

# src/cost/semantic_cache.py

class SemanticCache:
    def __init__(self, similarity_threshold=0.95, embedding_model=None):
        self.cache = []
        self.similarity_threshold = similarity_threshold
        self.embedding_model = embedding_model
    
    def find_similar(self, prompt: str) -> Optional[str]:
        """Find semantically similar cached prompt."""
        if not self.embedding_model:
            return None
        
        prompt_embedding = self.embedding_model.embed_text(prompt)
        
        for cached in self.cache:
            similarity = self.cosine_similarity(
                prompt_embedding,
                cached['embedding']
            )
            
            if similarity >= self.similarity_threshold:
                return cached['response']
        
        return None
    
    def cosine_similarity(self, a, b):
        """Calculate cosine similarity."""
        import numpy as np
        return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b))
    
    def add(self, prompt: str, response: str):
        """Add prompt to cache."""
        if self.embedding_model:
            embedding = self.embedding_model.embed_text(prompt)
        else:
            embedding = None
        
        self.cache.append({
            'prompt': prompt,
            'response': response,
            'embedding': embedding
        })

31.4.3 Prompt Caching

# src/cost/prompt_cache.py

class PromptCache:
    def __init__(self):
        self.cache = {}
    
    def cache_prompt(self, prompt_template: str, parameters: Dict):
        """Cache a prompt with parameters."""
        cache_key = self.generate_key(prompt_template, parameters)
        self.cache[cache_key] = {
            'template': prompt_template,
            'parameters': parameters,
            'timestamp': datetime.now()
        }
    
    def get_cached_prompt(self, prompt_template: str, 
                         parameters: Dict) -> Optional[str]:
        """Get cached prompt if available."""
        cache_key = self.generate_key(prompt_template, parameters)
        
        if cache_key in self.cache:
            cached = self.cache[cache_key]
            # Format the cached template with parameters
            return cached['template'].format(**cached['parameters'])
        
        return None
    
    def generate_key(self, template: str, parameters: Dict) -> str:
        """Generate cache key."""
        key_data = f"{template}:{json.dumps(parameters, sort_keys=True)}"
        return hashlib.md5(key_data.encode()).hexdigest()

# src/cost/batch_processor.py

class BatchProcessor:
    def __init__(self, model_client, batch_size=10):
        self.model_client = model_client
        self.batch_size = batch_size
    
    def process_batch(self, prompts: List[str]) -> List[str]:
        """Process multiple prompts in a batch."""
        results = []
        
        for i in range(0, len(prompts), self.batch_size):
            batch = prompts[i:i + self.batch_size]
            batch_results = self.model_client.generate_batch(batch)
            results.extend(batch_results)
        
        return results
    
    def estimate_savings(self, n_requests: int) -> Dict:
        """Estimate cost savings from batching."""
        # Individual processing cost
        individual_cost = n_requests * 0.002  # Example cost per request
        
        # Batch processing cost
        n_batches = (n_requests + self.batch_size - 1) // self.batch_size
        batch_cost = n_batches * 0.01  # Example cost per batch
        
        savings = individual_cost - batch_cost
        savings_percent = (savings / individual_cost * 100) if individual_cost > 0 else 0
        
        return {
            'individual_cost': individual_cost,
            'batch_cost': batch_cost,
            'savings': savings,
            'savings_percent': savings_percent
        }

class TieredModelStrategy:
    def __init__(self, model_router):
        self.model_router = model_router
    
    def execute(self, input_data: Dict) -> Dict:
        """Execute with tiered model strategy."""
        # Assess complexity
        complexity = self.assess_complexity(input_data)
        
        # Route to appropriate tier
        if complexity == 'low':
            model = 'gpt-3.5-turbo'
        elif complexity == 'medium':
            model = 'gpt-4-turbo'
        else:
            model = 'gpt-4'

        prompt = input_data['prompt']
        
        # Simple heuristics
        if len(prompt) < 100:
            return 'low'
        elif len(prompt) < 500:
            return 'medium'
        else:
            return 'high'

31.7.3 Pattern 3: Adaptive Caching

Adjust caching strategy based on usage patterns:

# src/patterns/adaptive_cache.py

class AdaptiveCache:
    def __init__(self, base_cache):
        self.base_cache = base_cache
        self.access_patterns = {}
    
    def get(self, prompt: str, model: str) -> Optional[str]:
        """Get from cache with adaptive strategy."""
        result = self.base_cache.get(prompt, model)
        
        # Track access pattern
        key = self.base_cache.get_cache_key(prompt, model)
        if key not in self.access_patterns:
            self.access_patterns[key] = 0
        self.access_patterns[key] += 1
        
        return result
    
    def should_cache(self, prompt: str, model: str) -> bool:
        """Determine if response should be cached."""
        # Cache if prompt is likely to be repeated
        key = self.base_cache.get_cache_key(prompt, model)
        access_count = self.access_patterns.get(key, 0)
        
        # Cache if accessed more than once
        return access_count > 0

31.9 Common Pitfalls

31.9.1 1. Over-Optimization

Problem: Sacrificing quality to reduce costs.

Solution: Maintain quality thresholds while optimizing.

31.9.2 2. Ignoring Hidden Costs

Problem: Focusing only on model costs, ignoring infrastructure and operations.

Solution: Consider total cost of ownership.

31.9.3 3. Stale Caching

Problem: Using outdated cached responses.

Solution: Implement proper cache invalidation strategies.

31.9.4 4. Inadequate Monitoring

Problem: Not tracking costs until they become problematic.

Solution: Implement cost monitoring from the start.

31.9.5 5. Premature Optimization

Problem: Optimizing before understanding usage patterns.

Solution: Monitor usage first, then optimize based on data.
